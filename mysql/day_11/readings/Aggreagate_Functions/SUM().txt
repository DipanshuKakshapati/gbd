--SUM()--

Returns the total sum of a numeric column.

Example:

SELECT SUM(Quantity)
FROM OrderDetails;


--with where--

SELECT SUM(Quantity)
FROM OrderDetails
WHERE ProductId = 11;

--with group by--

SELECT OrderID, SUM(Quantity) AS [Total Quantity]
FROM OrderDetails
GROUP BY OrderID;

--with expression--

SELECT SUM(Quantity * 10)
FROM OrderDetails;
